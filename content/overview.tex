\begin{frame}{设计原则}
    \begin{itemize}
      \item \alert{延迟计算}：图的构造与执行分离，并推迟计算图的执行过程
      \item \alert{原子OP}：OP是最小的抽象计算单元，支持构造复杂的网络模型
      \item \alert{抽象设备}：支持CPU, GPU, ASIC多种异构设备类型
      \item \alert{抽象任务}：基于Task的PS，支持优化算法和网络模型的扩展
    \end{itemize}
\end{frame}
